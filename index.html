  <b>Visa lager:</b><br>

  <input type="checkbox" id="kommun" checked> Kommuner<br>

  <input type="checkbox" id="naringsliv" checked> Näringsliv<br>

  <input type="checkbox" id="foreningar" checked> Föreningar<br>

  <input type="checkbox" id="projekt" checked> Projektaktörer<br>

<!-- Leaflet JS -->

<script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

<!-- MarkerCluster JS -->

<script src="https://unpkg.com/leaflet.markercluster/dist/leaflet.markercluster.js"></script>


<script>

  var map = L.map('map').setView([63.5, 11.0], 7);


  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: '© OpenStreetMap contributors'

  }).addTo(map);


  // Lagergrupper med klustring

  var kommunLayer = L.markerClusterGroup();

  var naringslivLayer = L.markerClusterGroup();

  var foreningarLayer = L.markerClusterGroup();

  var projektLayer = L.markerClusterGroup();


  // Exempeldata

  var data = [

    {name: "Östersund", type: "kommun", lat: 63.1792, lon: 14.6355, info: "Kommun i Jämtland"},

    {name: "Trondheim", type: "kommun", lat: 63.4305, lon: 10.3951, info: "Kommun i Trøndelag"},

    {name: "Företag X", type: "naringsliv", lat: 63.1792, lon: 14.6400, info: "Näringsliv Östersund"},

    {name: "Företag Y", type: "naringsliv", lat: 63.4300, lon: 10.4000, info: "Näringsliv Trondheim"},

    {name: "Förening A", type: "foreningar", lat: 63.1800, lon: 14.6300, info: "Förening i Östersund"},

    {name: "Projekt P", type: "projekt", lat: 63.4310, lon: 10.4100, info: "Projekt i Trondheim"}

  ];


  // Lägg till markörer i rätt lager

  data.forEach(function(d){

      var marker = L.marker([d.lat, d.lon]).bindPopup("<b>" + d.name + "</b><br>" + d.info);

      if(d.type === "kommun") kommunLayer.addLayer(marker);

      if(d.type === "naringsliv") naringslivLayer.addLayer(marker);

      if(d.type === "foreningar") foreningarLayer.addLayer(marker);

      if(d.type === "projekt") projektLayer.addLayer(marker);

  });


  // Lägg lager på kartan

  kommunLayer.addTo(map);

  naringslivLayer.addTo(map);

  foreningarLayer.addTo(map);

  projektLayer.addTo(map);


  // Funktion för checkboxar

  document.getElementById('kommun').addEventListener('change', function(e){

      if(this.checked) map.addLayer(kommunLayer); else map.removeLayer(kommunLayer);

  });

  document.getElementById('naringsliv').addEventListener('change', function(e){

      if(this.checked) map.addLayer(naringslivLayer); else map.removeLayer(naringslivLayer);

  });

  document.getElementById('foreningar').addEventListener('change', function(e){

      if(this.checked) map.addLayer(foreningarLayer); else map.removeLayer(foreningarLayer);

  });

  document.getElementById('projekt').addEventListener('change', function(e){

      if(this.checked) map.addLayer(projektLayer); else map.removeLayer(projektLayer);

  });

</script>
